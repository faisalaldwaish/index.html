def plot_chronic_diseases(df):
    """
    Calculate and plot the percentage of patients with chronic diseases.
    
    Args:
        df (pd.DataFrame): The dataframe containing the data.
    """
    # Calculate percentages
    chronic_diseases = {
        'Hypertension': (df['Hypertension'].sum() / len(df)) * 100,
        'Diabetes': (df['Diabetes'].sum() / len(df)) * 100,
        'Alcoholism': (df['Alcoholism'].sum() / len(df)) * 100,
        'Handicap': (df['Handicap'].sum() / len(df)) * 100,
    }
    
    # Plot pie chart
    plt.figure(figsize=(8, 8))
    plt.pie(
        chronic_diseases.values(),
        labels=chronic_diseases.keys(),
        autopct='%1.1f%%',
        startangle=140
    )
    plt.title('Percentage of Patients with Chronic Diseases')
    plt.axis('equal')
    plt.show()

# Call the function
plot_chronic_diseases(df)
